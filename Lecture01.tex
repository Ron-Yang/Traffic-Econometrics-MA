%##############################################################
\frame{\frametitle{Econometric models}
%##############################################################
Since econometrics describes things quantitatively, its basic language
is mathematics and its basic concept a \bfdef{(mathematical) model}


\maindm{
Y_k=f_k(\tilx_1, ..., \tilx_m, ..,\tilx_M, \beta_0, ..., \beta_j, ..., \beta_J)
+ \epsilon_k
= f_k(\tilvecx, \vecbeta) + \epsilon_k
}


\pause
\bi
\item A model consists of one or more equations $Y_k=f_k(\tilvecx,
  \vec{\beta}) + \epsilon_k$ explaining the output quantity $Y_k$ as a
  function of the input $\tilvecx$
\pause \item to \emph{tune} the model, there are \bfdef{model parameters}
  $\vec{\beta}$
\pause \item a model can be stochastic ($\epsilon_k \neq 0$) or deterministic
  ($\epsilon_k = 0$)
\pause \item the above formulation is the most general one and
includes \emph{all} conceivable econometric models. 
\ei
}

%##############################################################
\frame{\frametitle{Endogenous Variables}
%##############################################################

\bi
\item The name \bfdef{endogenous variable} comes from the Greek
  \emph{endo} ($\epsilon\nu\delta$)  and the 
  suffix \emph{gen} ($\gamma \epsilon\nu o\varsigma$) meaning
  generated from the inside [the model]
\pause\item in systems theory, the endogenous variables are the
\bfdef{output}
\pause\item mathematically, the endogenous variable are the
\bfdef{dependent variables}
\pause \item logically, they are the \bfdef{explained variables}
\ei
\pause \textit{Example mode choice}: $Y_k$: \#decisions for mode $k$ (e.g.,
walking, cycling, public transport, car, combined/others)

}

%##############################################################
\frame{\frametitle{Exogeneous variables}
%##############################################################

\bi
\item The name \bfdef{exogenous variable} comes from the Greek
  \emph{exo} ($\epsilon\xi\omega$)  and the 
  suffix \emph{gen} ($\gamma \epsilon\nu o\varsigma$) meaning
  ``coming from the outside''
\pause\item in systems theory, the endogenous variables are the
\bfdef{input}
\pause\item mathematically, the endogenous variable are the
\bfdef{independent variables}
\pause \item logically, they are the \bfdef{explanatory variables}
\pause \item an additive fixed function of explanatory variables is often
called a \bfdef{factor}
\ei
\pause Since the exogenous variable describe explanatory factors, they are
generally considered to be deterministic with all stochasticity
deferred to additional random terms $\epsilon_k$ 
\vspace{1em}

\pause \textit{Example mode choice}: $\tilde{x}_m$ can be travel times or costs for the
different modes, and also socioeconomic factors such as age, gender,
or income


}

%##############################################################
\frame{\frametitle{Random terms}
%##############################################################

The \bfdef{random terms}, also called \bfdef{residual terms} $\epsilon_k$
  (from \emph{residuum}: the rest) summarize all what is not known and
  cannot be explained by the exogenous/explanatory variables:
\vspace{-0.5em}

\maintextbox{0.55\textwidth}{\emph{Scio nescio} (I know that I do not know)}

\pause 
possible reasons for $\epsilon_k$:
\bi
\item cannot include all factors \bfred{(watch out for bias!)}
\pause \item the model is not appropriate (e.g., explaining the fuel
  consumption by a linear function of speed)
\pause\item the data used for model calibration contain errors
\pause\item in case of human decisions: 
\maintextbox{0.7\textwidth}{
man $\neq$ machine; \quad homo $\neq$ \emph{homo oeconomicus} }
\ei

\pause \textit{Example mode choice}: we ignored the weather or
 the additional utility to stop by
at a bakery (not provided by public transport), or need of a certain
mode (car) for subsequent trips of 
that day

}

%##############################################################
\frame{\frametitle{Model parameters}
%##############################################################
The \bfdef{model parameters} $\beta_j$, $j=0, \ldots, J$ \emph{tune} the
model to fit the data
\pause
\bi
\item The parameters are determined by fitting the model to
  \emph{learning data sets}, a process called \bfdef{calibration}
\pause \item To test the \emph{explanatory/prediction power} of a
model, the calibrated model is applied to \emph{test data sets} with
known output, a process called \bfdef{validation}
\pause \item In contrast to the exogenous variables changing from application
  to application, the parameters are \emph{fixed} after calibration.
\ei

\pause \maintextbox{0.65\textwidth}{The existence of well validated models is
  the \emph{raison d' \^{e}tre} for econometrics as such}

\pause \textit{Example mode choice}: Parameters characterize, e.g.,
the monetary \bfdef{value of time} (VoT) in \euro{}/h

}


%##############################################################
\frame{\frametitle{Model functions}
%##############################################################
The \bfdef{model function} is a mathematical representation of the
economic process under investigation. The model's mathematical
structure must reflect reality as well as 
  possible:

\bi
\pause \item linear \emph{vs.} nonlinear
\pause \item deterministic \emph{vs.} stochastic
\pause \item single or multiple equations that may be linked, chained, or
  with feedback
\pause \item which exogenous variables are relevant?
\ei
%
\pause
\maintextbox{0.65\textwidth}{The model structure defines the
  qualitative aspects and the model parameters the quantitative aspects
  of whatever is to be investigated}
\pause 
\maintextbox{0.65\textwidth}{\emph{Make it as simple as possible but
    not simpler}\\
\hspace*{\fill}(Einstein)} 
}

%##############################################################
\frame{\frametitle{Mathematical structure I: linear \emph{vs.} nonlinear}
%##############################################################
Four steps from linearity to nonlinearity:
\bi
\item[1.] \bfdef{Truly linear models:}
\bdm
Y=\hat{y}(\tilvecx, \vecbeta)+\epsilon
=\sum_{j=0}^J\beta_j\tilde{x}_j+\epsilon=\vecbeta\tr\tilvecx+\epsilon
\edm
Because of linearity, each endogenous variables has its own
uncoupled single equation, so it is enough to consider a single component.

\pause \item[2.] \bfdef{Parameter-linear (quasi-linear) models}
\bdm
Y=\hat{y}(\tilvecx, \vecbeta)+\epsilon
=\sum_{j=0}^J\beta_jg_j(\tilvecx)+\epsilon
=\sum_{j=0}^J\beta_jx_j+\epsilon
=\vecbeta\tr\vec{x}+\epsilon
\edm
If it is reasonable to assume fixed (generally nonlinear) functions
$x_j=g_j(\tilvecx)$, we 
have a linear model with the \bfdef{factors} $x_j$ becoming the
``new'' exogenous variables
\ei

}

%##############################################################
\frame{\frametitle{Example: average distance travelled in vehicles per year}
%##############################################################

\bi
\item $y$: distance [km/year] (deterministic because of
  ``average'')
\pause \item $\tilx_1$: income [\euro{}/year]
\pause \item $\tilx_2$: fuel cost [\euro{}/liter]
\pause \item Two exogenous variables $\to$ 4 factors: \\
$g_0(\tilvecx)=1$, 
 $g_1(\tilvecx)=\tilx_1$,
 $g_2(\tilvecx)=\tilx_2$,
 $g_3(\tilvecx)=\tilx_1 \tilx_2$,
\ei
\pause
\bdm
\text{\textbf{Model}:} \quad
y=\beta_0+\beta_1\tilx_1+\beta_2\tilx_2+\beta_3\tilx_1\tilx_2=\sum_{j=0}^3\beta_jx_j
\edm

\bi
\pause \itemAsk \colAsk{Discuss the elasticity 
$
\epsilon_2=\frac{\bar{x}_2}{\bar{y}}\abl{y}{x_2}=\frac{\beta_2\bar{x}_2}{y}=-0.15
$
}
\pause \itemAnswer {\colAnswer{\footnotesize \unit[0.15]{\%} 
decrease in kilometrage per increase of the fuel costs  by \unit[1]{\%}}}

\pause \itemAsk \colAsk{Discuss the meaning of the factors, particularly the
product $x_3$}
\pause \itemAnswer {\colAnswer{\footnotesize 
$x_0=1$: constant; $x_1=\tilx_1$: increase with income ($\beta_1>0$);
$x_2=\tilx_2$: price sensitivity ($\beta_2<0$);
$x_3=\tilx_1\tilx_2$: increase of price sensitivity (becoming less
    negative) with increasing income ($\beta_3>0$)
}}

\ei
}


%##############################################################
\frame{\frametitle{Mathematical structure I: linear \emph{vs.} nonlinear}
%##############################################################

\bi
\item[3.] \bfdef{Nonlinear models that can be linearized}

Classical example: unlimited growth
\bdm
G(t)=G_0e^{t/\tau+\epsilon}
\edm
\vspace{-1em}
\bi
\pause \item endogenous variable $G$: growth measure, e.g., company size,
  \#items sold of a newly introduced product
\pause \item exogenous variable $t$: time
\pause \item parameter $G_0$: initial growth measure
\item parameter $\tau$: time for growing by a factor of $e=2.71...$
\pause \item random multiplicative factor  $e^{\epsilon}$
\ei
\pause
\bdm
\text{\textbf{Linearisation}:} \quad
Y(t)=\ln G(t)=\ln G_0+\frac{t}{\tau}+\epsilon
\edm
\pause Reformulation  by setting
$x_0=1$, $x_1=t$, $\beta_0=\ln G_0$, $\beta_1=1/\tau$:
\bdm
\text{\textbf{Standard form}:} \quad
Y(\vec{x})=\vecbeta\tr\vec{x}+\epsilon
\edm
\ei
}
